
#define ll long long
class Solution {
public:
int n;
vector<bool> prime;
  void sieve(int n)
  {
      prime[0] = prime[1] = false;  
      for(int i = 2; i <= n; i++) 
      {   
          if (prime[i] && (ll)i * i <= n)
          {
              for(int j = i * i; j <= n; j += i) prime[j] = false;
          }
      }
  }
    
    vector<int> closestPrimes(int left, int right) {
        prime.resize(right+1, true);
        sieve(right+1);
        int prev = -1, curr = -1;
        vector<int> ans = {-1,-1};
        int mini = INT_MAX;
        for(int i=left; i<=right; i++)
        {
            if(prime[i] == 1)
            {
                if(curr == -1) curr = i;
                else{
                    prev = curr;
                    curr = i;
                    if(curr-prev < mini)
                    {
                        ans = {prev, curr};
                        mini = curr-prev;
                    }
                }
            }
        }
        return ans;
    }
};
